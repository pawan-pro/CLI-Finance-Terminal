# CLI Finance Terminal - Institutional Report Commands

## Generate Standard Daily Report
python -m src.cli.commands.report

## Generate Institutional-Grade Daily Report
python -m src.cli.commands.report --institutional

## Generate Report with Custom Save Directory
python -m src.cli.commands.report --save-dir ./my_reports

## Generate Institutional Report with Custom Directory
python -m src.cli.commands.report --institutional --save-dir ./my_institutional_reports

## Generate Text Format Report
python -m src.cli.commands.report --format txt

## Generate Institutional Text Report
python -m src.cli.commands.report --institutional --format txt

## Generate Report with Verbose Output
python -m src.cli.commands.report --verbose

## Generate Institutional Report with Verbose Output
python -m src.cli.commands.report --institutional --verbose

## Generate Report with Wine MT5 Integration (if available)
python -m src.cli.commands.report --wine-mt5

## Generate Institutional Report with Wine MT5 Integration
python -m src.cli.commands.report --institutional --wine-mt5

## View Help for Report Command
python -m src.cli.commands.report --help

## Run Institutional Report Test Script
python test_institutional_report.py

## Install Required Dependencies
pip install scipy scikit-learn seaborn

ls -la ./reports/
